    Les fonctions

    * fonctions : def

    .. runpython::
        :showcode:
        :toggle: out

        import cProfile

        nombres = [9, 7, 5, 4, 6, 7, 3, 1, 7, 8]

        def moyenne(ens):
            return sum(ens) / len(ens)

        def ecarttype(ens):
            var = [(n - moyenne(ens)) ** 2 for n in ens]
            return (sum(var) / len(var)) ** 0.5

        print(moyenne(nombres))
        print(ecarttype(nombres))

        with cProfile.Profile() as pr:
            for n in range(100000):
                ecarttype(nombres)

        pr.print_stats()
